// Use DBML to define your database structure
// Docs: https://dbml.dbdiagram.io/docs

Table Role {
  id integer [primary key]
  name text
}

Table User {
    id integer [primary key]
    name text
    email text UNIQUE
    telephone text
    password text
    role_id integer [ref: > Role.id]
    avatar_url text
}

Table ProjectStatus {
  id integer [primary key]
  name text
}

Table Event {
    id integer [primary key]
    name text
    start_date date
    end_date date
}

Table Project {
    event_id integer [ref: > Event.id, not null]
    id integer [primary key]
    status_id integer [ref: > Project_Status.id, not null]
    idea text
    description text
    team_name text
    team_avatar_url text
    iniator_id integer [ref: > User.id, not null]
    goal text
    components text
    skills text
}

Table Participant {
  id INTEGER [primary key]
  user_id integer [ref: > User.id, not null]
  project_id integer [ref: > Project.id, not null]
}
